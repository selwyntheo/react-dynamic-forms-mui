üíª Usage Example

import { DynamicForm } from 'react-dynamic-forms-mui';

const formConfig = {
  id: 'contact-form',
  title: 'Contact Us',
  sections: [{
    id: 'personal',
    fields: [{
      id: 'name',
      name: 'name',
      label: 'Full Name',
      type: 'text',
      required: true,
      validation: [
        { type: 'required', message: 'Name is required' }
      ]
    }]
  }]
};

function App() {
  return (
    <DynamicForm 
      config={formConfig}
      onSubmit={(data, isValid) => console.log(data)}
    />
  );
}
